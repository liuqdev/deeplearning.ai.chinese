def propagate(w, b, X, Y):
    """
    (cost)

    :
    w -- ,shape  (num_px * num_px * 3, 1)numpy array
    b -- , 
    X --  (num_px * num_px * 3, )
    Y --  ( 0 , 1 )  (1, number of examples)

    :
    cost -- (cost)
    dw -- w ,  w shape
    db -- b ,  b shape
    
    :
    - .np.log(), np.dot()
    """
    
    m = X.shape[1]
    
    # ( X  COST)
    ###  ### ( 2 )
    Z=np.dot(np.transpose(w),X)+b
    A = sigmoid(Z)                                  # 
    cost = -1*((np.sum((Y*np.log(A))+((1-Y)*(np.log(1-A))),axis=1))/m)                             #  cost
    ###  ###
    
    #  ()
    ###  ### ( 2 )
    dw = np.dot(X, np.transpose(A-Y))/m
    db = np.sum((A-Y), axis=1)/m
    ###  ###

    assert(dw.shape == w.shape)
    assert(db.dtype == float)
    cost = np.squeeze(cost)
    assert(cost.shape == ())
    
    grads = {"dw": dw,
             "db": db}
    
    return grads, cost

def optimize(w, b, X, Y, num_iterations, learning_rate, print_cost = False):
    """
     w  b.
    
    :
    w -- ,shape  (num_px * num_px * 3, 1)numpy array
    b -- , 
    X --  (num_px * num_px * 3, )
    Y --  ( 0 , 1 )  (1, number of examples)
    num_iterations -- 
    learning_rate -- ,
    print_cost -- True,loss
    
    :
    params -- , w   b
    grads -- , cost  w  b ()
    costs -- ,cost,
    
    :
    ,:
        1) , cost .propagate().
        2)  w  b.
    """
    
    costs = []
    
    for i in range(num_iterations):
        
        
        # Cost    ( 1-4 )
        ###  ### 
        grads, cost =  propagate(w, b, X, Y)
        ###  ###
        
        # 
        dw = grads["dw"]
        db = grads["db"]
        
        #  ( 2 )
        ### hi ###
        w = w-learning_rate*dw
        b = b-learning_rate*db
        ###  ###
        
        #  cost 
        if i % 100 == 0:
            costs.append(cost)
        
        # 100 cost
        if print_cost and i % 100 == 0:
            print (" %i  Cost: %f" %(i, cost))
    
    params = {"w": w,
              "b": b}
    
    grads = {"dw": dw,
             "db": db}
    
    return params, grads, costs

def predict(w, b, X):
    '''
     (w, b) , , 0  1
    
    :
    w -- ,shape  (num_px * num_px * 3, 1)numpy array
    b -- , 
    X --  (num_px * num_px * 3, )
    
    :
    Y_prediction -- numpy array (),X(0/1)
    '''
    
    m = X.shape[1]
    Y_prediction = np.zeros((1,m))
    w = w.reshape(X.shape[0], 1)
    
    #  "A" 
    ###  ### ( 1 )
    A = sigmoid(np.dot(w.T, X)+b)
    ###  ###
    
    for i in range(A.shape[1]):

        #  A[0,i]  p[0,i]
        ###  ### ( 4 )
        if A[0][i]>0.5 :
            Y_prediction[0][i]=1
        else:
            Y_prediction[0][i]=0
        ###  ###
    
    assert(Y_prediction.shape == (1, m))
    
    return Y_prediction

def model(X_train, Y_train, X_test, Y_test, num_iterations = 2000, learning_rate = 0.5, print_cost = False):
    """
    ,
    
    :
    X_train -- ,shape (num_px * num_px * 3, m_train)numpy array
    Y_train -- ,  numpy array (), shape (1, m_train)
    X_test -- ,  numpy array, shape (num_px * num_px * 3, m_test)
    Y_test -- ,  numpy array (), shape (1, m_test)
    num_iterations -- , 
    learning_rate -- , optimize()
    print_cost -- true,100cost
    
    :
    d -- ,.
    """
    
    ###  ###
    
    # 0 ( 1 )
    w, b = initialize_with_zeros(X_train.shape[0])
    
    #  ( 1 )
    parameters, grads, costs = optimize(w, b, X_train, Y_train, num_iterations, learning_rate, print_cost)

    # "parameters" w  b
    w = parameters["w"]
    b = parameters["b"]
    
    # / ( 2 )
    Y_prediction_test = predict(w, b, X_test)
    Y_prediction_train = predict(w, b,X_train )

    ###  ###

    # print((100 - np.mean(np.abs(Y_prediction_train - Y_train)) * 100))
    #  Errors
    print("train accuracy: {} %".format(100 - np.mean(np.abs(Y_prediction_train - Y_train)) * 100))
    print("test accuracy: {} %".format(100 - np.mean(np.abs(Y_prediction_test - Y_test)) * 100))

    
    d = {"costs": costs,
         "Y_prediction_test": Y_prediction_test, 
         "Y_prediction_train" : Y_prediction_train, 
         "w" : w, 
         "b" : b,
         "learning_rate" : learning_rate,
         "num_iterations": num_iterations}
    
    return d
